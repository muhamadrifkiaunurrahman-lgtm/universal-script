local function showNotifier(message)
	local playerGui = player:WaitForChild("PlayerGui")
	if playerGui:FindFirstChild("KiyNotifier") then
		playerGui.KiyNotifier:Destroy()
	end

	local screenGui = Instance.new("ScreenGui")
	screenGui.Name = "KiyNotifier"
	screenGui.Parent = playerGui

	local label = Instance.new("TextLabel")
	label.Size = UDim2.new(0, 300, 0, 45)
	label.Position = UDim2.new(0.5, -150, -0.1, 0)
	label.BackgroundColor3 = Color3.fromRGB(30, 30, 35)
	label.TextColor3 = Color3.fromRGB(0, 255, 200)
	label.TextStrokeTransparency = 0.8
	label.Font = Enum.Font.GothamBold
	label.TextScaled = true
	label.Text = message
	label.Parent = screenGui

	local corner = Instance.new("UICorner")
	corner.CornerRadius = UDim.new(0, 8)
	corner.Parent = label

	local padding = Instance.new("UIPadding")
	padding.PaddingBottom = UDim.new(0, 5)
	padding.PaddingTop = UDim.new(0, 5)
	padding.Parent = label

	local sound = Instance.new("Sound")
	sound.SoundId = "rbxassetid://9118820863"
	sound.Volume = 0.5
	sound.Parent = label
	sound:Play()

	local tweenIn = TweenService:Create(label, TweenInfo.new(0.4, Enum.EasingStyle.Back), {
		Position = UDim2.new(0.5, -150, 0.05, 0)
	})
	tweenIn:Play()

	task.delay(2, function()
		local tweenOut = TweenService:Create(label, TweenInfo.new(0.4, Enum.EasingStyle.Quad), {
			Position = UDim2.new(0.5, -150, -0.1, 0)
		})
		tweenOut:Play()
		tweenOut.Completed:Wait()
		screenGui:Destroy()
	end)
end

local dragging
local dragInput
local dragStart
local startPos

local function update(input)
	local delta = input.Position - dragStart
	MainFrame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
end

TopBar.InputBegan:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
		dragging = true
		dragStart = input.Position
		startPos = MainFrame.Position
		input.Changed:Connect(function()
			if input.UserInputState == Enum.UserInputState.End then
				dragging = false
			end
		end)
	end
end)

TopBar.InputChanged:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
		dragInput = input
	end
end)

UserInputService.InputChanged:Connect(function(input)
	if input == dragInput and dragging then
		update(input)
	end
end)

local function createButton(name, text, callback)
	local btn = Instance.new("TextButton")
	btn.Name = name
	btn.Text = text
	btn.BackgroundColor3 = Color3.fromRGB(40, 40, 45)
	btn.TextColor3 = Color3.fromRGB(255, 255, 255)
	btn.Font = Enum.Font.GothamBold
	btn.TextSize = 14
	btn.Parent = ScrollFrame

	local corner = Instance.new("UICorner")
	corner.CornerRadius = UDim.new(0, 6)
	corner.Parent = btn

	btn.MouseEnter:Connect(function()
		TweenService:Create(btn, TweenInfo.new(0.2), {BackgroundColor3 = Color3.fromRGB(60, 60, 70)}):Play()
	end)
	btn.MouseLeave:Connect(function()
		TweenService:Create(btn, TweenInfo.new(0.2), {BackgroundColor3 = Color3.fromRGB(40, 40, 45)}):Play()
	end)

	btn.MouseButton1Click:Connect(callback)
	return btn
end

local function createTextBox(name, placeholder, callback)
	local box = Instance.new("TextBox")
	box.Name = name
	box.PlaceholderText = placeholder
	box.Text = ""
	box.BackgroundColor3 = Color3.fromRGB(25, 25, 30)
	box.TextColor3 = Color3.fromRGB(0, 255, 200)
	box.Font = Enum.Font.Gotham
	box.TextSize = 14
	box.Parent = ScrollFrame

	local corner = Instance.new("UICorner")
	corner.CornerRadius = UDim.new(0, 6)
	corner.Parent = box

	local border = Instance.new("UIStroke")
	border.Color = Color3.fromRGB(60, 60, 70)
	border.ApplyStrokeMode = Enum.ApplyStrokeMode.Border
	border.Parent = box

	box.FocusLost:Connect(function(enterPressed)
		if enterPressed and box.Text ~= "" then
			callback(box.Text)
			box.Text = ""
		end
	end)
	return box
end

local espEnabled = false
local espObjects = {}

createButton("ESPBtn", "Toggle ESP", function()
	espEnabled = not espEnabled
	if espEnabled then
		for _, obj in ipairs(workspace:GetDescendants()) do
			if obj:IsA("Humanoid") and obj.Parent ~= player.Character then
				local hl = Instance.new("Highlight")
				hl.Adornee = obj.Parent
				hl.FillColor = Color3.fromRGB(255, 0, 0)
				hl.OutlineColor = Color3.fromRGB(255, 255, 255)
				hl.Parent = obj.Parent
				table.insert(espObjects, hl)
			end
		end
		showNotifier("ESP Activated")
	else
		for _, hl in pairs(espObjects) do
			if hl and hl.Parent then hl:Destroy() end
		end
		espObjects = {}
		showNotifier("ESP Deactivated")
	end
end)

local flying = false
local bodyVel
createButton("FlyBtn", "Toggle Fly", function()
	flying = not flying
	local char = player.Character
	if flying and char and char:FindFirstChild("HumanoidRootPart") then
		bodyVel = Instance.new("BodyVelocity")
		bodyVel.MaxForce = Vector3.new(9e9, 9e9, 9e9)
		bodyVel.Parent = char.HumanoidRootPart
		showNotifier("Fly Activated (Use WASD)")
	else
		if bodyVel then bodyVel:Destroy() end
		showNotifier("Fly Deactivated")
	end
end)

local infJump = false
UserInputService.JumpRequest:Connect(function()
	if infJump and player.Character and player.Character:FindFirstChild("Humanoid") then
		player.Character.Humanoid:ChangeState(Enum.HumanoidStateType.Jumping)
	end
end)

createButton("InfJumpBtn", "Infinite Jump", function()
	infJump = not infJump
	showNotifier(infJump and "Infinite Jump On" or "Infinite Jump Off")
end)

local noclip = false
RunService.Stepped:Connect(function()
	if noclip and player.Character then
		for _, part in pairs(player.Character:GetDescendants()) do
			if part:IsA("BasePart") then part.CanCollide = false end
		end
	end
end)

createButton("NoclipBtn", "Toggle Noclip", function()
	noclip = not noclip
	showNotifier(noclip and "Noclip On" or "Noclip Off")
end)

createButton("GodModeBtn", "God Mode (Heal)", function()
	if player.Character and player.Character:FindFirstChild("Humanoid") then
		player.Character.Humanoid.Health = player.Character.Humanoid.MaxHealth
		showNotifier("Health Restored to Max")
	end
end)

createTextBox("WalkSpeedBox", "Set WalkSpeed (Enter)", function(val)
	local num = tonumber(val)
	if num and player.Character and player.Character:FindFirstChild("Humanoid") then
		player.Character.Humanoid.WalkSpeed = num
		showNotifier("WalkSpeed set to " .. num)
	end
end)

createTextBox("JumpPowerBox", "Set JumpPower (Enter)", function(val)
	local num = tonumber(val)
	if num and player.Character and player.Character:FindFirstChild("Humanoid") then
		player.Character.Humanoid.UseJumpPower = true
		player.Character.Humanoid.JumpPower = num
		showNotifier("JumpPower set to " .. num)
	end
end)

createTextBox("GotoBox", "Teleport to Player", function(val)
	local target = Players:FindFirstChild(val)
	if target and target.Character and target.Character:FindFirstChild("HumanoidRootPart") then
		if player.Character and player.Character:FindFirstChild("HumanoidRootPart") then
			player.Character.HumanoidRootPart.CFrame = target.Character.HumanoidRootPart.CFrame + Vector3.new(0, 3, 0)
			showNotifier("Teleported to " .. target.Name)
		end
	else
		showNotifier("Player not found!")
	end
end)

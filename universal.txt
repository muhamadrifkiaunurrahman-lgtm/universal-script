local drag, dragInp, dragStart, startPos
top.InputBegan:Connect(function(i)
	if i.UserInputType == Enum.UserInputType.MouseButton1 then
		drag = true
		dragStart = i.Position
		startPos = main.Position
		i.Changed:Connect(function() if i.UserInputState == Enum.UserInputState.End then drag = false end end)
	end
end)
top.InputChanged:Connect(function(i)
	if i.UserInputType == Enum.UserInputType.MouseMovement then dragInp = i end
end)
UIS.InputChanged:Connect(function(i)
	if i == dragInp and drag then
		local delta = i.Position - dragStart
		main.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
	end
end)

local function mkTab(name)

	btn.BackgroundTransparency = 1
	btn.Text = name
	btn.TextColor3 = Color3.fromRGB(200, 200, 200)
	btn.Font = Enum.Font.GothamBold
	btn.TextSize = 14
	btn.Parent = tabCon
	local sf = Instance.new("ScrollingFrame")
	sf.Size = UDim2.new(1, 0, 1, 0)
	sf.BackgroundTransparency = 1
	sf.ScrollBarThickness = 4
	sf.Visible = false
	sf.Parent = pageCon
	local gl = Instance.new("UIGridLayout")
	gl.CellSize = UDim2.new(0.48, 0, 0, 35)
	gl.CellPadding = UDim2.new(0.04, 0, 0, 10)
	gl.Parent = sf
	table.insert(pages, {b = btn, p = sf})
	btn.MouseButton1Click:Connect(function()
		for _, x in pairs(pages) do
			x.p.Visible = false
			x.b.TextColor3 = Color3.fromRGB(200, 200, 200)
		end
		sf.Visible = true
		btn.TextColor3 = themeCol
	end)
	return sf
end

local function updTheme(c)
	themeCol = c
	for _, x in pairs(tObjs) do if x.o and x.o.Parent then x.o[x.prop] = c end end
	for _, x in pairs(pages) do if x.p.Visible then x.b.TextColor3 = c end end
end

local function mkUI(type, par, txt, cb)
	if type == "Btn" then
		local b = Instance.new("TextButton")
		b.Text = txt
		b.BackgroundColor3 = Color3.fromRGB(40, 40, 45)
		b.TextColor3 = Color3.fromRGB(255, 255, 255)
		b.Font = Enum.Font.GothamBold
		b.TextSize = 12
		b.Parent = par
		Instance.new("UICorner", b).CornerRadius = UDim.new(0, 6)
		b.MouseButton1Click:Connect(cb)
		return b
	elseif type == "Box" then
		local b = Instance.new("TextBox")
		b.PlaceholderText = txt
		b.Text = ""
		b.BackgroundColor3 = Color3.fromRGB(25, 25, 30)
		b.TextColor3 = themeCol
		table.insert(tObjs, {o = b, prop = "TextColor3"})
		b.Font = Enum.Font.Gotham
		b.TextSize = 12
		b.Parent = par
		Instance.new("UICorner", b).CornerRadius = UDim.new(0, 6)
		local s = Instance.new("UIStroke")
		s.Color = Color3.fromRGB(60, 60, 70)
		s.ApplyStrokeMode = Enum.ApplyStrokeMode.Border
		s.Parent = b
		b.FocusLost:Connect(function(ep) if ep and b.Text ~= "" then cb(b.Text) b.Text = "" end end)
		return b
	end
end

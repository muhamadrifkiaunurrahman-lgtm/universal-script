local drag, dragInp, dragStart, startPos
top.InputBegan:Connect(function(i)
	if i.UserInputType == Enum.UserInputType.MouseButton1 then
		drag = true
		dragStart = i.Position
		startPos = main.Position
		i.Changed:Connect(function() if i.UserInputState == Enum.UserInputState.End then drag = false end end)
	end
end)
UIS.InputChanged:Connect(function(i)
	if drag and i.UserInputType == Enum.UserInputType.MouseMovement then
		local delta = i.Position - dragStart
		main.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
	end
end)

local pages = {}
local function mkTab(name, icon)
	local b = Instance.new("TextButton")
	b.Size = UDim2.new(1, 0, 0, 40)
	b.BackgroundColor3 = Color3.fromRGB(25, 25, 30)
	b.Text = "  " .. icon .. "  " .. name
	b.TextColor3 = Color3.fromRGB(200, 200, 200)
	b.Font = Enum.Font.GothamMedium
	b.TextSize = 14
	b.TextXAlignment = Enum.TextXAlignment.Left
	b.Parent = tabCon
	Instance.new("UICorner", b).CornerRadius = UDim.new(0, 8)

	local s = Instance.new("ScrollingFrame")
	s.Size = UDim2.new(1, 0, 1, 0)
	s.BackgroundTransparency = 1
	s.ScrollBarThickness = 2
	s.Visible = false
	s.Parent = pageCon
	Instance.new("UIGridLayout", s).CellSize = UDim2.new(0.48, 0, 0, 38)

	table.insert(pages, {b = b, p = s})
	b.MouseButton1Click:Connect(function()
		for _, x in pairs(pages) do x.p.Visible = false x.b.TextColor3 = Color3.fromRGB(200, 200, 200) end
		s.Visible = true b.TextColor3 = neon
	end)
	return s
end

local function mkUI(type, par, txt, cb)
	local f = Instance.new("Frame")
	f.BackgroundColor3 = Color3.fromRGB(25, 25, 30)
	f.Parent = par
	Instance.new("UICorner", f).CornerRadius = UDim.new(0, 6)

	if type == "Tog" then
		local t = Instance.new("TextLabel")
		t.Size = UDim2.new(1, -40, 1, 0)
		t.Position = UDim2.new(0, 10, 0, 0)
		t.Text = txt
		t.TextColor3 = Color3.fromRGB(230, 230, 230)
		t.Font = Enum.Font.Gotham
		t.TextSize = 12
		t.TextXAlignment = Enum.TextXAlignment.Left
		t.BackgroundTransparency = 1
		t.Parent = f
		local b = Instance.new("TextButton")
		b.Size = UDim2.new(0, 24, 0, 24)
		b.Position = UDim2.new(1, -34, 0.5, -12)
		b.BackgroundColor3 = Color3.fromRGB(40, 40, 45)
		b.Text = ""
		b.Parent = f
		Instance.new("UICorner", b).CornerRadius = UDim.new(0, 6)
		local check = Instance.new("TextLabel")
		check.Size = UDim2.new(1,0,1,0)
		check.Text = "✓"
		check.TextColor3 = neon
		check.Visible = false
		check.BackgroundTransparency = 1
		check.Parent = b
		local on = false
		b.MouseButton1Click:Connect(function() on = not on check.Visible = on cb(on) end)
	elseif type == "Box" then
		local b = Instance.new("TextBox")
		b.Size = UDim2.new(1, -20, 1, -10)
		b.Position = UDim2.new(0, 10, 0, 5)
		b.PlaceholderText = txt
		b.Text = ""
		b.TextColor3 = neon
		b.Font = Enum.Font.Gotham
		b.TextSize = 12
		b.BackgroundTransparency = 1
		b.Parent = f
		b.FocusLost:Connect(function(ep) if ep and b.Text ~= "" then cb(b.Text) b.Text = "" end end)
	end
end

local aimbot = false
RS.RenderStepped:Connect(function()
	if aimbot then
		local target = nil local dist = 1000
		for _, pl in pairs(Players:GetPlayers()) do
			if pl ~= p and pl.Character and pl.Character:FindFirstChild("Head") then
				local pos, vis = cam:WorldToViewportPoint(pl.Character.Head.Position)
				if vis then
					local mag = (Vector2.new(mouse.X, mouse.Y) - Vector2.new(pos.X, pos.Y)).Magnitude
					if mag < dist then dist = mag target = pl.Character.Head end
				end
			end
		end
		if target then cam.CFrame = CFrame.new(cam.CFrame.Position, target.Position) end
	end
end)
mkUI("Tog", combTab, "Aimbot Camera", function(on) aimbot = on end)
